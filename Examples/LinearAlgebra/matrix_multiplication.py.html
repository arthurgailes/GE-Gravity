4    def matrix_multiplication(a, b): 
5        """ 
6         Calculates the matrix product of A and b. 
7        :param a: The m*n matrix. 
8        :param b: The n*1 vector. 
9        :return: The matrix product defined as A * b. 
10       """ 
11        
12       result = [None] * len(a) 
13    
14       for i in range(len(a)): 
15           sum = 0.0 
16    
17           for j in range(len(b)): 
18               sum += a[i][j] * b[j] 
19    
20           result[i] = sum 
21    
22       return result
